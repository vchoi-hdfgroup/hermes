116  void SetupAdapterMode() {
117  char* adapter_mode_str = getenv(kAdapterMode);
118  if (adapter_mode_str == nullptr) {
119  // default is Persistent mode
120  adapter_mode = AdapterMode::kDefault;
121  } else {
122  if (strcmp(kAdapterDefaultMode, adapter_mode_str) == 0) {
123  adapter_mode = AdapterMode::kDefault;
124  } else if (strcmp(kAdapterBypassMode, adapter_mode_str) == 0) {
125  adapter_mode = AdapterMode::kBypass;
126  } else if (strcmp(kAdapterScratchMode, adapter_mode_str) == 0) {
127  adapter_mode = AdapterMode::kScratch;
128  } else {
129  // TODO(hari): @errorhandling throw error.
130  return;
131  }
132  }
133  char* adapter_paths_str = getenv(kAdapterModeInfo);
134  std::vector<std::string> paths_local;
135  if (adapter_paths_str) {
136  paths_local = StringSplit(adapter_paths_str, kPathDelimiter);
137  }
138  adapter_paths = paths_local;
139  }
140 
141  bool Persists(FILE* fh) { return Persists(GetFilenameFromFP(fh)); }
142  bool Persists(int fd) { return Persists(GetFilenameFromFD(fd)); }

144  bool Persists(std::string path) {
145  if (adapter_mode == AdapterMode::kDefault) {
146  if (adapter_paths.empty()) {
147  return true;
148  } else {
149  for (const auto& pth : adapter_paths) {
150  if (path.find(pth) == 0) {
151  return true;
152  }
153  }
154  return false;
155  }
156  } else if (adapter_mode == AdapterMode::kScratch) {
157  if (adapter_paths.empty()) {
158  return false;
159  } else {
160  for (const auto& pth : adapter_paths) {
161  if (path.find(pth) == 0) {
162  return false;
163  }
164  }
165  return true;
166  }
167  }
168  return true;
169  }
