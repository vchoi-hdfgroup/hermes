81  void InitializeHermes(bool is_mpi = false) {
82  if (ref == 0) {
83  this->is_mpi = is_mpi;
84  char* async_flush_mode = getenv(kHermesAsyncFlush);
85 
86  if (async_flush_mode && async_flush_mode[0] == '1') {
87  global_flushing_mode = FlushingMode::kAsynchronous;
88  } else {
89  global_flushing_mode = FlushingMode::kSynchronous;
90  }
91 
92  char* hermes_config = getenv(kHermesConf);
93 
94  if (this->is_mpi) {
95  MPI_Comm_rank(MPI_COMM_WORLD, &rank);
96  MPI_Comm_size(MPI_COMM_WORLD, &comm_size);
97  // TODO(chogan): Need a better way to distinguish between client and
98  // daemon. https://github.com/HDFGroup/hermes/issues/206
99  if (comm_size > 1) {
100  hermes = hermes::InitHermesClient(hermes_config);
101  } else {
102  this->is_mpi = false;
103  hermes = hermes::InitHermesDaemon(hermes_config);
104  }
105  } else {
106  hermes = hermes::InitHermesDaemon(hermes_config);
107  }
108  INTERCEPTOR_LIST->SetupAdapterMode();
109  }
110  ref++;
111  }
116  void FinalizeHermes() {
117  if (ref == 1) {
118  if (this->is_mpi) {
119  MPI_Barrier(MPI_COMM_WORLD);
120  char *stop_daemon = getenv(kStopDaemon);
121  bool shutdown_daemon = true;
122 
123  if (stop_daemon && stop_daemon[0] == '0') {
124  shutdown_daemon = false;
125  }
126 
127  hermes->FinalizeClient(shutdown_daemon);
128  } else {
129  hermes->Finalize(true);
130  }
131  }
132  ref--;
133  }
