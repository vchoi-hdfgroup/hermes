233 static bool IsBoFunction(const char * func_name) {
234  bool result = false;
235  int i = 0;
236 
237  while (func_name && *func_name != '\0' && i < kBoPrefixLength) {
238  if (func_name[i] != kBoPrefix[i]) {
239  break;
240  }
241  ++i;
242  }
243 
244  if (i == kBoPrefixLength) {
245  result = true;
246  }
247 
248  return result;
249 }

251 template<typename ReturnType, typename... Ts>
252 ReturnType RpcCall(RpcContext *rpc, u32 node_id, const char *func_name,
253  Ts... args) {
254  VLOG(1) << "Calling " << func_name << " on node " << node_id
255  << " from node " << rpc->node_id << std::endl;
256  ClientThalliumState *state = GetClientThalliumState(rpc);
257  bool is_bo_func = IsBoFunction(func_name);
258  std::string server_name = GetServerName(rpc, node_id, is_bo_func);
259 
260  if (is_bo_func) {
261  func_name += kBoPrefixLength;
262  }
263 
264  tl::remote_procedure remote_proc = state->engine->define(func_name);
265  // TODO(chogan): @optimization We can save a little work by storing the
266  // endpoint instead of looking it up on every call
267  tl::endpoint server = state->engine->lookup(server_name);
268 
269  if constexpr(std::is_same<ReturnType, void>::value) {
270  remote_proc.disable_response();
271  remote_proc.on(server)(std::forward<Ts>(args)...);
272  } else {
273  ReturnType result = remote_proc.on(server)(std::forward<Ts>(args)...);
274 
275  return result;
276  }
277 }
