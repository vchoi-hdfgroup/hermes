477 template<typename T>
478 std::vector<SwapBlob> PutToSwap(SharedMemoryContext *context, RpcContext *rpc,
479  BucketID id,
480  std::vector<std::vector<T>> &blobs,
481  std::vector<std::string> &names) {
482  size_t num_blobs = blobs.size();
483  std::vector<SwapBlob> result(num_blobs);
484 
485  for (size_t i = 0; i < num_blobs; ++i) {
486  SwapBlob swap_blob = PutToSwap(context, rpc, names[i], id,
487  (const u8*)blobs[i].data(),
488  blobs[i].size() * sizeof(T));
489  result.push_back(swap_blob);
490  }
491 
492  return result;
493 }
